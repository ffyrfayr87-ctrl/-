import time from flask import Flask, render_template_string, abort, request, make_response app = Flask(__name__) # --- نظام حماية بسيط ضد التخمين والطلبات المتكررة السريعة (Rate Limiter) --- ip_requests = {} @app.before_request def rate_limiter(): client_ip = request.remote_addr current_time = time.time() if client_ip in ip_requests: last_request_time = ip_requests[client_ip] # إذا كان الفارق بين الطلب والآخر أقل من 0.5 ثانية، يتم حظر الطلب مؤقتاً if current_time - last_request_time < 0.5: return abort(429, description="طلب سريع جداً! تم حظرك مؤقتاً لحماية الموقع من التخمين.") ip_requests[client_ip] = current_time # --- إضافة رؤوس أمان لحماية المتصفح من الاختراق والغش --- @app.after_request def add_security_headers(response): # لمنع تضمين الموقع داخل iFrame (حماية ضد Clickjacking) response.headers['X-Frame-Options'] = 'DENY' # لمنع المتصفح من تخمين نوع الملفات (حماية ضد Content Sniffing) response.headers['X-Content-Type-Options'] = 'nosniff' # تفعيل حماية المتصفح ضد حاقني الأكواد XSS response.headers['X-XSS-Protection'] = '1; mode=block' return response # --- التصميم والأكواد (HTML + CSS + JS) في ملف واحد --- HTML_TEMPLATE = """ آية الكرسي الإلكترونية

استماع إلى آية الكرسي

اضغط على الأيقونة للتشغيل
""" @app.route('/') def index(): return render_template_string(HTML_TEMPLATE) if __name__ == '__main__': # تشغيل السيرفر محلياً على البورت 5000 app.run(host='0.0.0.0', port=5000, debug=False)
